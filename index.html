function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { try { const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); const action = e.parameter.action; if (action === 'load') { const data = sheet.getDataRange().getValues(); const rows = data.slice(1).filter(row => row[0] && row[0] !== ''); return ContentService.createTextOutput(JSON.stringify({ success: true, data: rows })) .setMimeType(ContentService.MimeType.JSON); } else if (action === 'save') { const dados = JSON.parse(e.parameter.dados); // CABEÇALHOS CORRETOS (10 colunas) const headers = ['ID', 'NOME', 'CPF', 'WHATSAPP', 'VALORPARCELA', 'TOTALPARCELAS', 'PARCELAATUAL', 'CARTAO', 'DATAINICIO', 'PAGAMENTOS']; // Verifica/Adiciona cabeçalho const firstRow = sheet.getRange(1, 1, 1, headers.length).getValues()[0]; const hasHeader = firstRow[0] === 'ID'; if (!hasHeader) { sheet.getRange(1, 1, 1, headers.length).setValues([headers]); } // LIMPA TODOS OS DADOS (mantém cabeçalho) const lastRow = sheet.getLastRow(); if (lastRow > 1) { sheet.getRange(2, 1, lastRow - 1, headers.length).clearContent(); } // PREPARA OS DADOS NA ORDEM CORRETA const rows = dados.map(d => { // Garante que a data seja apenas YYYY-MM-DD let dataInicio = d.inicio; if (dataInicio && dataInicio.includes('T')) { dataInicio = dataInicio.split('T')[0]; } return [ d.id, // ID d.nome, // NOME d.cpf, // CPF d.whats, // WHATSAPP d.valorParcela, // VALORPARCELA d.totalParcelas, // TOTALPARCELAS d.parcelaAtual, // PARCELAATUAL d.cartao, // CARTAO dataInicio, // DATAINICIO JSON.stringify(d.pagamentos || []) // PAGAMENTOS ]; }); // ESCREVE OS DADOS if(rows.length > 0) { sheet.getRange(2, 1, rows.length, rows[0].length).setValues(rows); } return ContentService.createTextOutput(JSON.stringify({ success: true })) .setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput(JSON.stringify({ success: false, error: "Ação inválida" })) .setMimeType(ContentService.MimeType.JSON);
